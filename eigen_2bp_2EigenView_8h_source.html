37 template <typename T, int N, int C, typename XprKind_, int Rows_, int Cols_>
38 class FromBoostPython< EigenView<T,N,C,XprKind_,Rows_,Cols_> > {
39 public:
40 
41  BOOST_STATIC_ASSERT( N == 1 || N == 2 );
42 
43  BOOST_STATIC_ASSERT( N != 1 || Rows_ == 1 || Cols_ == 1 );
44 
45  explicit FromBoostPython(boost::python::object const & input) : _impl(input) {}
46 
47  bool convertible() {
48  try {
49  boost::numpy::ndarray array = boost::python::extract<boost::numpy::ndarray>(_impl.input);
50  if (N == 2) {
51  if (Rows_ == 1) {
52  array = array.reshape(boost::python::make_tuple(1, -1));
53  } else if (Cols_ == 1) {
54  array = array.reshape(boost::python::make_tuple(-1, 1));
55  }
56  if (Rows_ != Eigen::Dynamic && array.shape(0) != Rows_) return false;
57  if (Cols_ != Eigen::Dynamic && array.shape(1) != Cols_) return false;
58  } else if (N == 1) {
59  array = array.squeeze();
60  int requiredSize = Rows_ * Cols_;
61  if (requiredSize != Eigen::Dynamic && array.shape(0) != requiredSize) return false;
62  }
63  _impl.input = array;
64  if (!_impl.convertible()) return false;
65  } catch (boost::python::error_already_set) {
66  boost::python::handle_exception();
67  PyErr_Clear();
68  return false;
69  }
70  return true;
71  }
72 
73  EigenView<T,N,C,XprKind_,Rows_,Cols_> operator()() {
74  return EigenView<T,N,C,XprKind_,Rows_,Cols_>(_impl());
75  }
76 
77 private:
78  FromBoostPython< Array<T,N,C> > _impl;
79 };
